    ```python
    import csv
    import pymongo
    import dateutil.parser

    # MongoDBに接続
    #client = pymongo.MongoClient("mongodb://localhost:27017/")
    #client = pymongo.MongoClient("mongodb://admin:[pw]@localhost:27017/")
    client = pymongo.MongoClient("mongodb://[username]:[pw]@localhost:27017/")
    # データベースを選択
    db = client["[DB_name]"]
    # コレクションを選択
    collection = db["Collection_name"]

    # CSVファイルを開く
    with open("data.csv", "r") as f:
        # CSVリーダーを作成
        reader = csv.reader(f)
        # ドキュメントのリストを作成
        docs = []
        # 列名のリストを作成（例として、col1からcol55までの名前を付ける）
        col_names = ["col" + str(i) for i in range(1, 56)]
        # CSVファイルの各行に対して
        for row in reader:
            # ドキュメントを作成
            doc = {}
            # 列名と値を対応させてドキュメントに追加
            for col_name, value in zip(col_names, row):
                # 値が数値の場合はfloatに変換
                try:
                    value = float(value)
                # 値が日付の場合はDate型に変換
                except ValueError:
                    try:
                        value = dateutil.parser.parse(value)
                    # 値が文字列の場合はそのまま
                    except ValueError:
                        pass
                # ドキュメントに追加
                doc[col_name] = value
            # ドキュメントのリストに追加
            docs.append(doc)
        # bulk insertを実行
        collection.insert_many(docs)
    ```
